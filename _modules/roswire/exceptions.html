    def __init__(self, retcode: int, stdout: str) -> None:
        self.retcode = retcode
        self.stdout = stdout
        msg = "unexpected failure during bag playback [return code: {}]:\n{}"
        msg = msg.format(retcode, stdout)
        super().__init__(msg)

    """ROSWire found two or more types for a service, topic, or action."""

    entity: str
    existing: str
    conflicting: str

    def __str__(self) -> str:
        return (f"error declaring type {self.conflicting} for {self.entity}. "
